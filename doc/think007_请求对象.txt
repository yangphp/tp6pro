

1、引用请求对象方式

a、静态调用 
use think\facade\Request;

Request::param('name');

b、助手函数 推荐使用助手函数

request()->param('name');

2、获取当前的url

获取完整URL地址 不带域名
Request::url()

获取完整URL地址 带域名
Request::url(true)

3、获取当前的控制器和操作,多应用模式下获取不到

//获取当前控制器--获取不到什么鬼
echo Request::controller();

//获取当前操作--获取不到什么鬼
echo request()->action();
